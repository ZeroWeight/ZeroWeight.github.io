\entry{Research project during the 2023 internship in NVIDIA RAPIDS team, mentor Dr. Xiaoyun Wang, Dr. Joe Eaton}

\subsection*{3D Structured Molecule Generation with Equivariant Diffusion Models}
\denseouterlist{
\entry{Continue project after NVIDIA internship with Dr. Xiaoyun Wang et. al.. }
\entry{Using equivariant diffusion models for position generation and discrete diffusion models for atom type generation.}
\entry{Introducing the atom neutralization method which improves the molecule stability from $\sim 90\% \rightarrow 95\%$}
\entry{Presented at \href{https://www.nvidia.com/en-us/on-demand/session/gtcspring23-s52380/}{Nvidia GTC 2023: 3D Molecule Generation with the Equivariant Score-Based Diffusion Process}}
\entry{Accepted by ICML 2023 Workshop on Structured Probabilistic Inference \& Generative Modeling}}

\subsection*{Molecule Property Prediction with 3D Structured Data and Graph Neural Networks}
\denseouterlist{
\entry{Research project 2022 during the internship in NVIDIA RAPIDS team, mentor Dr. Xiaoyun Wang, Dr. Joe Eaton}
\entry{Using Contrastive Learning to learn the graph representation of 3D structured data with Equivariant Transformers}
\entry{Apply methods (Laplacian Embedding, Self-attention Pooling, Dropout, etc.) to improve the scalability of the GNNs}
\entry{Improve $50\%$ by contrastive learning compared with baseline model (accuracy (MAE) $0.0156 \rightarrow 0.0144$)}
}
\subsection*{AI-based Categorization to Decipher Reaction Mechanisms from Cyclic Voltammetry}
\denseouterlist{
\entry{Launched the project in August 2021 Predict the Cyclic Voltammetry mechanism using deep learning.}
